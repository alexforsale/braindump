Specific Variable#

std::env::var returns a Result<String, VarError>.

use std::env;

fn main () {
    let xdg_session_type = env::var("XDG_SESSION_TYPE");

    match xdg_session_type {
        Ok(val) =>  println!("XDG_SESSION_TYPE: {}", val),
        Err(e) => eprintln!("Couldn't read XDG_SESSION_TYPE: {}", e),
    }
}
